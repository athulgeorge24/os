#include <stdio.h>
#include <stdlib.h>
#include <time.h>

#define SIZE 3  // Buffer can hold 3 items

int buffer[SIZE];
int count = 0; // Current number of items in buffer
int in = 0;    // Index for producer
int out = 0;   // Index for consumer

void producer(int item) {
    buffer[in] = item;
    in = (in + 1) % SIZE;
    count++;
    printf("Produced %d \n", item);
}

void consumer() {
    int item = buffer[out];
    out = (out + 1) % SIZE;
    count--;
    printf("Consumed %d \n", item);
}

int main() {
    srand(time(NULL));
    int total_actions = 15; 

    for(int i = 0; i < total_actions; i++) {
        int num = rand() % 10;
        int choice = rand() % 2;

        if (choice == 0) { 
            if (count < SIZE) {
                producer(num);
            } else {
                // Buffer full, must consume
                consumer();
            }
        } 
        else {
            if (count > 0) {
                consumer();
            } else {
                // Buffer empty, must produce
                producer(num);
            }
        }
    }
    return 0;
}
